rm(list = ls())

## Loading required package: data.table
## Warning: package 'data.table' was built under R version 3.6.3
library("data.table")

#carrega dados
tabela <- read.csv("tabela.csv",sep=";")

#monta tabela com dados de malria apenas
ocor = tabela$ocorrencia[tabela$cod_zoo==11] #ocorrencias
an = tabela$ano[tabela$cod_zoo==11] #ano
re = tabela$cod_regiao[tabela$cod_zoo==11] #regiao
escola = tabela$escolaridade[tabela$cod_zoo==11] #escolaridade
munextr = tabela$mun_extr_pobreza[tabela$cod_zoo==11] #municipio de extrema pobreza
idade = tabela$classif_etaria[tabela$cod_zoo==11] #idade
gen = tabela$sexo[tabela$cod_zoo==11] #genero
evol = tabela$evolucao[tabela$cod_zoo==11] #evoluo do caso
desp <- data.table(an,ocor,re,escola,munextr,idade,gen,evol) #tabela com dados especficos de malria

Ocorrncias totais ao longo dos anos:

graf = tapply(desp$ocor,desp$an,sum)
bp <- barplot(graf, border = F, las=2, ylim = c(0,1300), xlab = "Anos", ylab = "Ocorrncias de malria", main = "Malria no Brasil ao longo dos anos", col = "DeepSkyBlue", cex.names = 0.6, cex.axis = 0.6, cex.main = 1.0, cex.lab = 0.7) #grfico de barras
text(x=as.vector(bp), y=graf+1, labels = graf, pos=3, offset = 0.05, cex = 0.6)
grid(ny = 6, nx = NA) #linhas de grade
abline(lsfit(1:length(graf),graf), col = "darkgreen") #linha de tendncia

    As ocorrncias de malria vm aumentando com o passar dos anos. Isto pode ser um indicativo de necessidade de melhoria em aes de preveno, como o combate ao mosquito transmissor, por exemplo.
