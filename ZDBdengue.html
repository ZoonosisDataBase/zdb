rm(list = ls())

# carrega pacote de manipulação de tabelas
if (!require("data.table")){
  install.packages("data.table")
}
## Loading required package: data.table
## Warning: package 'data.table' was built under R version 3.6.3
library("data.table")

#carrega dados
tabela <- read.csv("tabela.csv",sep=";")

#monta tabela com dados de dengue apenas
ocor = tabela$ocorrencia[tabela$cod_zoo==2] #ocorrencias
an = tabela$ano[tabela$cod_zoo==2] #ano
re = tabela$cod_regiao[tabela$cod_zoo==2] #regiao
escola = tabela$escolaridade[tabela$cod_zoo==2] #escolaridade
munextr = tabela$mun_extr_pobreza[tabela$cod_zoo==2] #municipio de extrema pobreza
idade = tabela$classif_etaria[tabela$cod_zoo==2] #idade
gen = tabela$sexo[tabela$cod_zoo==2] #genero
evol = tabela$evolucao[tabela$cod_zoo==2] #evolução do caso
desp <- data.table(an,ocor,re,escola,munextr,idade,gen,evol) #tabela com dados específicos de dengue

Ocorrências totais ao longo dos anos:

graf = tapply(desp$ocor,desp$an,sum)
bp <- barplot(graf, border = F, las=2, ylim = c(0,60000), xlab = "Anos", ylab = "Ocorrências de dengue", main = "Dengue no Brasil ao longo dos anos", col = "DeepSkyBlue", cex.names = 0.6, cex.axis = 0.6, cex.main = 1.0, cex.lab = 0.7) #gráfico de barras
text(x=as.vector(bp), y=graf+1, labels = graf, pos=3, offset = 0.05, cex = 0.6)
grid(ny = 6, nx = NA) #linhas de grade
abline(lsfit(1:length(graf),graf), col = "darkgreen") #linha de tendência

    Após o início dos surtos de dengue no país e a estabilização no número de casos a partir de 2017, não foram mais computados dados de ocorrência da doença no DATASUS, mesmo tratando-se de uma das zoonoses que mais afeta o país em termos de número de casos.
