library("data.table")

#carrega dados
tabela <- read.csv("tabela.csv",sep=";")

#monta tabela com dados de tuberculose apenas
ocor = tabela$ocorrencia[tabela$cod_zoo==15] #ocorrencias
an = tabela$ano[tabela$cod_zoo==15] #ano
re = tabela$cod_regiao[tabela$cod_zoo==15] #regiao
escola = tabela$escolaridade[tabela$cod_zoo==15] #escolaridade
munextr = tabela$mun_extr_pobreza[tabela$cod_zoo==15] #municipio de extrema pobreza
idade = tabela$classif_etaria[tabela$cod_zoo==15] #idade
gen = tabela$sexo[tabela$cod_zoo==15] #genero
evol = tabela$evolucao[tabela$cod_zoo==15] #evoluo do caso
desp <- data.table(an,ocor,re,escola,munextr,idade,gen,evol) #tabela com dados especficos de tuberculose

Ocorrncias totais ao longo dos anos:

graf = tapply(desp$ocor,desp$an,sum)
bp <- barplot(graf, border = F, las=2, ylim = c(0,32000), xlab = "Anos", ylab = "Ocorrncias de tuberculose", main = "Tuberculose no Brasil ao longo dos anos", col = "DeepSkyBlue", cex.names = 0.6, cex.axis = 0.6, cex.main = 1.0, cex.lab = 0.7) #grfico de barras
text(x=as.vector(bp), y=graf+1, labels = graf, pos=3, offset = 0.05, cex = 0.6)
grid(ny = 6, nx = NA) #linhas de grade
abline(lsfit(1:length(graf),graf), col = "darkgreen") #linha de tendncia

    A tuberculose uma das zoonoses com maior quantidade e detalhamento de dados no sistema do DATASUS. De 2015 a 2018 o nmero de casos manteve-se em um mesmo patamar, havendo reduo em 2019. Esta pode ser decorrente tanto de aes de preveno quanto da ausncia de dados totais disponveis at o momento. A ltima atualizao em massa do DATASUS ocorreu em Maro de 2020, sendo que em nota ressalta-se que os dados ainda precisam de reviso.
