#limpa dados
rm(list = ls())

setwd("insira o diretorio em que seu arquivo foi armazenado")

# verificar diretório
getwd()

# carrega pacote de manipulação de tabelas
if (!require("data.table")){
  install.packages("data.table")
}
## Loading required package: data.table
## Warning: package 'data.table' was built under R version 3.6.3
library("data.table")

#carrega dados
tabela <- read.csv("tabela.csv",sep=";")

#monta tabela com dados de leishmaniose visceral apenas
ocor = tabela$ocorrencia[tabela$cod_zoo==8] #ocorrencias
an = tabela$ano[tabela$cod_zoo==8] #ano
re = tabela$cod_regiao[tabela$cod_zoo==8] #regiao
escola = tabela$escolaridade[tabela$cod_zoo==8] #escolaridade
munextr = tabela$mun_extr_pobreza[tabela$cod_zoo==8] #municipio de extrema pobreza
idade = tabela$classif_etaria[tabela$cod_zoo==8] #idade
gen = tabela$sexo[tabela$cod_zoo==8] #genero
evol = tabela$evolucao[tabela$cod_zoo==8] #evolução do caso
desp <- data.table(an,ocor,re,escola,munextr,idade,gen,evol) #tabela com dados específicos de leishmaniose visceral

Ocorrências totais ao longo dos anos:

graf = tapply(desp$ocor,desp$an,sum)
bp <- barplot(graf, border = F, las=2, ylim = c(0,5200), xlab = "Anos", ylab = "Ocorrências de leishmaniose visceral", main = "Leishmaniose visceral no Brasil ao longo dos anos", col = "DeepSkyBlue", cex.names = 0.6, cex.axis = 0.6, cex.main = 1.0, cex.lab = 0.7) #gráfico de barras
text(x=as.vector(bp), y=graf+1, labels = graf, pos=3, offset = 0.05, cex = 0.6)
grid(ny = 6, nx = NA) #linhas de grade
abline(lsfit(1:length(graf),graf), col = "darkgreen") #linha de tendência

    A leishmaniose tegumentar não possui ocorrências registradas no ano de 2019. De 2015 a 2018 o número de casos manteve-se em um patamar alto e lentamente progressivo o que indica necessidade de maior atenção aos cuidados em prevenção da doença.


Ocorrências totais por região:

    Percebe-se que existe um número alto de casos em bebês e/ou crianças (escolaridade “Não se aplica”), que também pode ser confirmado pelo gráfico das classificações etárias.

    Embora não se manifeste com o maior número, como no caso da maioria das doenças, ainda há uma presença muito elevada de dados não preenchidos nas fichas de notificação. Estes aparecem como “Em branco/Ignorado” e analisando em termos de escolaridade dos casos, este fator torna mais difícil a elaboração de estratégias adequadas de prevenção.
